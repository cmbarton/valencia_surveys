Basis for Temporal Index (Temporal Rank or TR in data files)MS FoxPro program (series of If/Then statements in XBase)Uses information on artifact classes for each collection unit (subsector)procedure temprank** does all procedures in programdo lndo endo lupdo eupdo mpdo pleistholoreturn* Late Neolithicprocedure lngo topdo while not eof()If ln_tools>0 or left(ceram_date,2)="N2"	replace lneol with .9	else		if (ceramics+grndstone>0) and bkd_tools=0 and en_tools=0			repl eneol with .7				if neol_tools+ceramics+grndstone>0					replace lneol with .5					else						if blade_tech>0							replace lneol with .3							else								if lithics+grndstone+ceramics>0									replace lneol with .1									else replace lneol with 0								endif							endif				endif		endifendifskipenddoreturn* Early Neolithicprocedure engo topdo while not eof()If (bkd_tools>0 and (ceramics>0 or en_tools>0) and ln_tools=0 and left(ceram_date,2)<>"N2") or left(ceram_date,2)="N1"	replace eneol with .9	else		if (neol_tools+ceramics+grndstone>0) and ln_tools=0 and left(ceram_date,2)<>"N2""			replace eneol with .7			else				if bkd_tools+ceramics>0					replace eneol with .5					else						if blade_tech>0							replace eneol with .3							else								if lithics+grndstone+ceramics>0									replace eneol with .1									else replace eneol with 0								endif						endif				endif		endifendifskipenddoreturn* Late Upper Paleolithic - Mesolithicprocedure lupgo topdo while not eof()If bkd_tools>0 and ceramics+ln_tools+en_tools+neol_tools=0	replace LUPaleo with .9	else		if  bkd_tools>0 and cer_dens<75			replace LUPaleo with .7			else				if (bkd_tools+blade_tech>0) and cer_dens<75					replace LUPaleo with .5					else						if bkd_tools+blade_tech>0							replace LUPaleo with .3							else								if lithics>0									replace LUPaleo with .1									else replace LUPaleo with 0								endif						endif				endif		endifendifskipenddoreturn* Early Upper Paleolithicprocedure eupgo topdo while not eof()If UP_tools>0 and blade_tech>0 and bkd_tools+ceramics+ln_tools+en_tools+neol_tools=0	replace UPaleo with .9	else		if  (UP_tools+blade_tech>0) and bkd_tools+ceramics+ln_tools+en_tools+neol_tools=0			replace UPaleo with .7			else				if  (UP_tools+blade_tech>0) and cer_dens<75					replace UPaleo with .5					else						if UP_tools+blade_tech>0							replace UPaleo with .3							else								if lithics>0									replace UPaleo with .1									else replace UPaleo with 0								endif						endif				endif		endifendifskipenddoreturn* Middle Paleolithicprocedure mpgo topdo while not eof()If MP_tools>0 and  UP_tools+blade_tech+bkd_tools+ceramics+ln_tools+neol_tools=0	MPaleo=.9	else		if  MP_tools>0 and bkd_tools+ceramics+ln_tools+neol_tools+en_tools=0			replace MPaleo with .7			else				if  MP_tools>0 and cer_dens<75					replace MPaleo with .5					else						if  MP_tools+flake_tech>0							replace MPaleo with .3							else								if lithics>0									replace MPaleo with .1									else replace MPaleo with 0								endif						endif				endif		endifendifskipenddoreturnprocedure pleistholorepl all pleist with max(mpaleo,UPaleo)repl all holo with max(lupaleo,eneol,lneol)return* Late Neol. tools:  Bifacial projectile points and denticulated sickle blades.;* The latter also occus in Bronze Age contexts, but is only found in association;* with other Neolithic artifacts in our collections for the Polop; hence it is;* used as a marker for the Late Neolithic.* Neol. tools: Plain (undenticulated) sickle blades and bifacial drills (taladros);* Backed tools: Backed tools include backed bladelets, backed points, and geometric;* microliths.* EUP tools: Upper Paleolithic tools include endscrapers, burins, truncations,;* and perforators.* M. Paleo tools: Middle Paleolithic tools include sidescrapers (including;* transverse and déjéte scrapers), notch/denticulates, and Mousterian points.* Blade tech.: Besides prismatic blades, evidence for blade technology includes;* crest blades, blade cores, and blade core face and platform rejuvination flakes.  Settlement Intensity Index (Settlement Index or SI in data files)1.	Settlement weight (Settl_wt) = artifact density for each collection unit ranked according to all units in Polop Alto survey area2.	Settlement weight for each unit given in ntile values: 0, 0.10, 0.25, 0.50, 0.75, 0.90, 1.00 to indicate ntile ranking3.	[TI]_SI (e.g., MPaleo_SI or LNeol_SI) = Temporal Index multiplied by settlement weight for each unit.4.	[TI]_SI values ranked in ntiles ([TI]si_ntile. 	Only the those at median and above values calculate.	[TI]si_ntile values calcuated in ranked groups of 0.90-1.00, 0.75-0.90, and 0.50-0.75
